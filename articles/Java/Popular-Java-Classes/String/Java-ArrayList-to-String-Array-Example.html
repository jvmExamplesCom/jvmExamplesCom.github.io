Java ArrayList to String Array Example

The following example shows how to convert an ArrayList of strings to string array in Java. 

          
           
import java.util.ArrayList;

public class ConvertArrayListToStringArrayExample {

 public static void main(String[] args) {

  //Create an ArrayList object
  ArrayList<String> arrayList = new ArrayList<String>();

  //Add elements to Arraylist
  arrayList.add("Java");
  arrayList.add("C++");
  arrayList.add("PHP");

  //get the ArrayList size
  int size = arrayList.size();

  //create a string array with the same size as arraylist
  String[] strArray = new String[size];

  //copy the contents of ArrayList to string array
  arrayList.toArray(strArray);

  //display the array content
  for(int i=0; i<strArray.length; i++) {
   System.out.println(strArray[i]);
  }
 }
}
