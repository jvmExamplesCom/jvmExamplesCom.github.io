public class JavaPyramid2Example { 
   public static void main(String[] args) { 
      // Create pyramid using a for loop 
      for (int i = 0; i < 5; i++) { 
         for (int j = 0; j <= i; j++) { 
            System.out.print("*" + " "); 
         } 
         System.out.println(); 
      } 
                   
      // Create pyramid using a while loop 
      int x = 4; 
      while (x >= 0) { 
         int y = 0; 
         while (y <= x) { 
            System.out.print("*" + " "); 
            y++; 
         } 
         System.out.println(); 
         x--; 
      } 
   }
