try {
    // Create a Cipher object
    Cipher cipher = Cipher.getInstance("AES/CBC/PKCS5Padding");

    // Create CipherInputStream and CipherOutputStream objects
    CipherOutputStream cos = new CipherOutputStream(new FileOutputStream("encrypted_file.txt"), cipher);
    CipherInputStream cis = new CipherInputStream(new FileInputStream("input_file.txt"), cipher);

    int nread;

    }

    // Close streams

    cos.close();

    System.out.println("Error: " + e.getMessage());
}
