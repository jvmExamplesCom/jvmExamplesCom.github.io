FrameLayout.LayoutParams lp = new FrameLayout.LayoutParams(300, 200, Gravity.TOP);

lp.setMarginStart(int start);
lp.setMarginEnd(int end);

lp.setMarginBottom(int bottom);

lp.setLayoutWeight(float weight);
