  • Attach the AbsListView to the ViewGroup.

<!-- Create a new ViewGroup with the desired layout parameters -->
ViewGroup mViewGroup = new ViewGroup(context);
LayoutParams layoutParams = new LayoutParams(LayoutParams.MATCH_PARENT, LayoutParams.MATCH_PARENT);
mViewGroup.setLayoutParams(layoutParams);

<!-- Instantiate a new AbsListView object -->
AbsListView mAbsListView = new AbsListView(context);

mAbsListView.setLayoutParams(layoutParams);

mViewGroup.addView(mAbsListView);
