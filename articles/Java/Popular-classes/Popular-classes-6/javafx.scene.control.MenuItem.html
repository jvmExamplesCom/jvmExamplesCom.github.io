MenuItem myMenuItem = new MenuItem("My Menu Item");

MenuItem

myMenuItem.setTooltip(new Tooltip("My Tooltip"));

MenuItems

menu.getItems().add(myMenuItem);

menu.getItems().addAll(myMenuItem);
