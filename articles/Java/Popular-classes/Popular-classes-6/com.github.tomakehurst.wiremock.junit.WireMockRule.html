What is Com.github.tomakehurst.wiremock.junit.WireMockRule

@Rule
public WireMockRule wireMockRule = new WireMockRule(options().dynamicPort());

@Test

    verify(getRequestedFor(urlEqualTo("/some/thing")));
}

verify
