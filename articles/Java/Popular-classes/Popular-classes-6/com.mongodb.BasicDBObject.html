com.mongodb.BasicDBObject in Java

BasicDBObject obj = new BasicDBObject();

Map<String, Object> pairs = new HashMap<>();

pairs.put("address", "123 Main St.");

BasicDBObject obj = new BasicDBObject(pairs);
