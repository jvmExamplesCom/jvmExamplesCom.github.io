        }
 
        serverSocket.close();
    }
}

        DataInputStream inputStream = new DataInputStream(socket.getInputStream());
        DataOutputStream outputStream = new DataOutputStream(socket.getOutputStream());

        System.out.println(response);
 
        outputStream.writeUTF("shutdown");

        socket.close();
    }
}
