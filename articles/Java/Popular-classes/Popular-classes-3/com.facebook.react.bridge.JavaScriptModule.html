com.facebook.react.bridge.JavaScriptModule
is a JavaScript interface used in React Native to enable communication between native code and JavaScript components.   public void exampleMethod(String value) {

  }

const { NativeModules } = require('react-native');

ExampleModule.exampleMethod("example value");
