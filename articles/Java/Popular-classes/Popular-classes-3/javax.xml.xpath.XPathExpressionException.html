javax.xml.xpath.XPathExpressionException

javax.xml.xpath.XPathExpressionException is an exception that is thrown when a problem occurs while parsing an XPath expression. It typically occurs when a malformed or invalid XPath expression is used in the code. This exception is a subclass of the org.xml.sax.SAXException and it extends the standard Java exception class java.lang.Exception.

import javax.xml.xpath.XPathConstants;

import javax.xml.xpath.XPathExpressionException;
import javax.xml.xpath.XPathFactory;

public class App

    public static void main( String[] args )

import javax.xml.xpath.XPathFactory;

public class App 
{
    public static void main( String[] args )
    {
        XPathFactory factory = XPathFactory.newInstance();
        XPath xPath = factory.newXPath();

        // invalid XPath expression
        String expression = "//*[@id='@id']";

        // check if the expression is valid
        boolean isValid = xPath.isValid(expression);

        if(isValid) 
		{
            // compile the XPath expression
            XPathExpression xPathExpression = xPath.compile(expression);

            Object result = xPathExpression.evaluate(document, XPathConstants.NODESET);
        } 
		else 
		{

    }
}
