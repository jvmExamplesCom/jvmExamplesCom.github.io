Creating a new 

          
           
DefaultMutableTreeNode

DefaultMutableTreeNode rootNode = new DefaultMutableTreeNode("Root");

add()
method: 

          
           
DefaultMutableTreeNode child1 = new DefaultMutableTreeNode("Child 1");
rootNode.add(child1);
