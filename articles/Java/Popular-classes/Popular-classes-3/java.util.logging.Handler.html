java.util.logging.Handler

java.util.logging.Handler

java.util.logging.Handler

java.util.logging.Logger

Logger logger = Logger.getLogger("MyLogger");
Handler logHandler = new Handler(logger);

logHandler.publish(myLogRecord);

logHandler.flush();
