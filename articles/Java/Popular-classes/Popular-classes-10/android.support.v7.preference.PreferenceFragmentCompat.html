Android's PreferenceFragmentCompat class provides an interface to manage settings within an Android app. This makes it easy to create settings with specific preferences, control the behavior of the settings, and access settings from outside the application.     // Add preferences
  }
}

MySettingsFragment settingsFragment = new MySettingsFragment();

public void onCreatePreferences(Bundle bundle, String s) {
