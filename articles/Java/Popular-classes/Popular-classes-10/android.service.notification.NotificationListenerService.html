The 

          
           
android.service.notification.NotificationListenerService

This class can also be used to post new notifications or cancel existing ones.

onListenerConnected()

onListenerDisconnected()

		// Code to execute when the listener is connected

		// Code to execute when the listener is disconnected
	}
}
