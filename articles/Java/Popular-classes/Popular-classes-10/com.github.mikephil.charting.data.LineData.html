LineData in Java

LineData data = new LineData();

data.addDataSet(new LineDataSet(someDataList));

BarChart chart = new BarChart(data);

chart.setBackgroundColor(Color.LTGRAY);

LinearLayout layout = (LinearLayout) findViewById(R.id.my_layout);
