        // Get the InputSplit object
        FileSplit fileSplit = (FileSplit) split;
        // Get the path of the file
        Path path = fileSplit.getPath();
        // Open the file
        FSDataInputStream in = fs.open(path);

    public boolean nextKeyValue() throws IOException {

            return true;
        }
        return false;
    }
    
    @Override

        return key;
    }

    }
