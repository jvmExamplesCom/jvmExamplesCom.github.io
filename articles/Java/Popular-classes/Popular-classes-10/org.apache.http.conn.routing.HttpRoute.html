org.apache.http.conn.routing.HttpRoute

HttpRoute(InetAddress target)

HttpRoute route = new HttpRoute(HttpHost("example.com"));

Routing Connections

HttpClient.route(HttpHost target, HttpRoute route, HttpContext context);
