Creating a StringRedisSerializer instance

StringRedisSerializer stringRedisSerializer = new StringRedisSerializer();

Using StringRedisSerializer

redisTemplate.setValueSerializer(stringRedisSerializer);

String someString = "Some String value";
byte[] serializedString = stringRedisSerializer.serialize(someString);
redisTemplate.opsForValue().set("key", serializedString);
