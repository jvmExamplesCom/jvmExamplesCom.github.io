Creating a ColorMatrixColorFilter

    1f, 0f, 0f, 0f, 0f,

    0f, 0f, 1.25f, 0f, 0f,
    0f, 0f, 0f, 1f, 0f
};
Paint paint = new Paint();
ColorMatrixColorFilter filter = new ColorMatrixColorFilter(colorMatrix);
paint.setColorFilter(filter);
