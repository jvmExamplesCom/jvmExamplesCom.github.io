org.springframework.security.oauth2.config.annotation.configurers.ClientDetailsServiceConfigurer

The org.springframework.security.oauth2.config.annotation.configurers.ClientDetailsServiceConfigurer is a class in the Spring Security OAuth 2 framework that provides methods to configure the client details services and token stores on an OAuth 2 authorization server. This class helps developers define the details of the client applications that they want to enable access to their resources. It also helps configure the type of tokens (JWT or plain) used to authenticate the client applications.

@Configuration
public class WebSecurityConfig extends WebSecurityConfigurerAdapter {

    @Override
    public void configure(HttpSecurity http) throws Exception {
        // code goes here 
    }

    @Override
    protected void configure(final ClientDetailsServiceConfigurer clientDetailsServiceConfigurer) throws Exception {
        clientDetailsServiceConfigurer
            .inMemory()
            .withClient("clientId")
            .secret("secret")
            .redirectUris("http://localhost:8080/callback")
            .authorizedGrantTypes("password", "refresh_token")
            .resourceIds("tokenInfo")
            .scopes("all");
    }
}

@Configuration
@EnableAuthorizationServer
public class OAuth2AuthorizationServerConfig extends AuthorizationServerConfigurerAdapter {

    @Override
    public void configure(final ClientDetailsServiceConfigurer clientDetailsServiceConfigurer) throws Exception {
        clientDetailsServiceConfigurer
            .inMemory()
            .withClient("clientId")
            .secret("secret")
            .redirectUris("http://localhost:8080/callback")
            .authorizedGrantTypes("password", "refresh_token")
            .resourceIds("tokenInfo")
            .scopes("all");
    }

    @Override
    public void configure(AuthorizationServerEndpointsConfigurer endpoints) throws Exception {
        endpoints
            .tokenStore(tokenStore())
            .accessTokenConverter(accessTokenConverter());
    }
}
