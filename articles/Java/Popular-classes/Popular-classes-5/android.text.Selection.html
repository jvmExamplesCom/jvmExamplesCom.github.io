android.text.Selection
class contains several methods for selecting text within an Editable or Spannable string. public static void selectAll(Editable text) 
public static void setSelection(Editable text, int start, int stop) 
public static final void extendSelection(Editable text, int index)

// Select all text
Selection.selectAll(myEditText.getText());
