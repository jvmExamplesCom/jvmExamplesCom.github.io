SSLSocketFactoryConfigurer configurer = new SSLSocketFactoryConfigurer();
SSLSocketFactory factory = new SSLSocketFactory(configurer);

URL url = new URL("https://www.example.com");
HttpsURLConnection conn = (HttpsURLConnection)url.openConnection(factory);
