javax.swing.event.TreeSelectionListener is an interface in the Java Swing library used to detect when the selection in a tree of nodes changes.     // Create a TreeModel with two nodes

    tree.addTreeSelectionListener(new MyTreeSelectionListener());
  }
}

  @Override
  public void valueChanged(TreeSelectionEvent e) {

  }
